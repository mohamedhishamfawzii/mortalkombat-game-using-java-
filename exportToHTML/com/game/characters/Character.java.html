package com.game.characters; 
 
import com.game.powers.Powers; 
 
import javax.swing.*; 
import java.awt.*; 
import java.awt.event.KeyEvent; 
import java.util.ArrayList; 
import java.util.Vector; 
 
public abstract class Character implements Characterizable { 
    protected int health; 
    protected ArrayList<Powers> power; 
    protected int xp; 
    protected ImageIcon icon; 
    private static Vector<String> characters = new Vector<String>(); 
    private int x,dx; 
    private int y=200; 
 
 
    protected Image charimg; 
    protected String imgname; 
 
 
    public Character(int health, ArrayList<Powers> power, int xp, ImageIcon icon) { 
        this.health = health; 
        this.power = power; 
        this.xp = xp; 
        this.icon = icon; 
        loadImage(); 
        x=60; 
    }

    public int getX(){ 
        return x; 
    } 
    public int getY(){ 
        return y; 
    } 
 
    public void move(){ 
        x+=dx; 
    } 
 
    /** 
     * this function set array of vectors by the names of available character 
     * 
     * @return Vector String 
     */ 
    public static Vector<String> getCharacters() { 
        characters.add("Sub Zero"); 
        characters.add("Scorpion"); 
        characters.add("Rose"); 
        characters.add("Retsuu"); 
        characters.add("Akuma"); 
 
        return characters; 
    } 
 
    public void loadImage(){ 
        ImageIcon im = new ImageIcon(getClass().getName()); 
        charimg = im.getImage(); 
    } 
 
    public int getHealth() { 
        return health; 
    } 
 
    public int getXp() { 
        return xp; 
    } 
 
    public ImageIcon getIcon() { 
        return icon; 
    } 
 
    public Image getImg() { 
        return charimg; 
    } 
}
